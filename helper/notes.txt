CREATE TABLE tasks (
    id SERIAL PRIMARY KEY,
    name TEXT NOT NULL,
    description TEXT,

    -- Период повторения
    period_type TEXT NOT NULL CHECK (period_type IN ('none','week','month','year')),

    -- Логика внутри периода
    schedule_type TEXT NOT NULL CHECK (schedule_type IN ('fixed','floating','none')),

    -- Для fixed-weekday (например, вторник)
    weekday INT CHECK (weekday BETWEEN 1 AND 7),

    -- Для fixed-date
    day_of_month INT CHECK (day_of_month BETWEEN 1 AND 31),
    month_of_year INT CHECK (month_of_year BETWEEN 1 AND 12),

    last_completed_at TIMESTAMP,  -- время последнего выполнения
    priority INT DEFAULT 0
);

Пояснение полей:

Поле	Для чего
period_type	Определяет период повторения: неделя, месяц, год
schedule_type	fixed = привязано к дню, floating = выполнить в любой день периода, none = однократная задача
weekday	Если fixed-weekday (например, вторник)
day_of_month, month_of_year	Для фиксированных дат (например, 15 марта)
last_completed_at	Для расчёта, была ли задача выполнена в текущем периоде
priority	Для сортировки между задачами

from datetime import datetime

def is_task_active(task, today=datetime.now()):
    # Однократная задача
    if task.period_type == 'none' or task.schedule_type == 'none':
        return task.last_completed_at is None

    # Задачи в пределах недели
    if task.period_type == 'week':
        current_week = today.isocalendar()[1]
        last_week = task.last_completed_at.isocalendar()[1] if task.last_completed_at else -1

        if task.schedule_type == 'floating':
            return current_week != last_week

        if task.schedule_type == 'fixed':
            today_weekday = today.isoweekday()
            return today_weekday == task.weekday and current_week != last_week

    # Задачи в пределах месяца
    if task.period_type == 'month':
        current_month = today.month
        last_month = task.last_completed_at.month if task.last_completed_at else -1

        if task.schedule_type == 'floating':
            return current_month != last_month

        if task.schedule_type == 'fixed':
            return today.day == task.day_of_month and current_month != last_month

    # Задачи в пределах года (fixed-date)
    if task.period_type == 'year':
        current_year = today.year
        last_year = task.last_completed_at.year if task.last_completed_at else -1
        return (today.day == task.day_of_month and today.month == task.month_of_year and current_year != last_year)

    return False
